\section{Scope And Boundary}
\label{sec:org134b9ea}
\begin{itemize}
\item In scope:
\begin{itemize}
\item input validation against canonical normalized input
\item astrology calculations
\item Human Design calculations
\item Gene Keys derivation
\item application of pinned mappings, thresholds, ordering, and formatting
\item structured success and error responses
\item deterministic metadata for reproducibility
\end{itemize}
\item Out of scope:
\begin{itemize}
\item birthplace search, geocoding, and timezone determination
\item UI and presentation behavior
\item persistence and operational logging
\item narrative, content, recommendations, coaching, and translation
\item any calculation or output not explicitly enumerated in the canon
\end{itemize}
\item Supported domains:
\begin{itemize}
\item Astrology
\item Human Design
\item Gene Keys
\end{itemize}
\item No separate "Trinity derived calculations" category exists in v1.
\item Engine boundary rule:
\begin{itemize}
\item input reaches the engine only after normalization
\item raw birthplace strings, unresolved timezone data, free-text dates/times, and ambiguous location data are outside scope
\end{itemize}
\end{itemize}

\item IANA Time Zone Database is the authoritative timezone standard.
\item Gregorian calendar is the authoritative civil calendar basis.
\item Tropical zodiac is authoritative across domains.
\item Placidus is the authoritative astrology house system.
\item Node policy by domain:
\begin{itemize}
\item Astrology uses Mean Node.
\item Human Design uses True Node.
\item Gene Keys inherits the Human Design basis.
\end{itemize}
\item External libraries are implementation vehicles only; library defaults are non-authoritative unless canon explicitly adopts them.
\item Third-party platforms may be used only for sanity checking, not as truth.
\end{itemize}

\section{Input Contract}
\label{sec:org2cbbb99}
\subsection{Canonical Payload}
\label{sec:org1d558b1}
\begin{itemize}
\item One person per payload.
\item Required fields:
\begin{center}
\begin{tabular}{lll}
field & type & format / rule\\
\hline
birth\textsubscript{date} & string & YYYY-MM-DD\\
birth\textsubscript{time} & string & HH:MM, 24-hour, minute only\\
timezone & string & IANA timezone identifier\\
latitude & number & decimal degrees, -90.0..90.0\\
longitude & number & decimal degrees, -180.0..180.0\\
\end{tabular}
\end{center}
\item Canonical logical shape:
\end{itemize}
\begin{verbatim}
{
  "birth_date": "YYYY-MM-DD",
  "birth_time": "HH:MM",
  "timezone": "Area/Location",
  "latitude": 0.000000,
  "longitude": 0.000000
}
\end{verbatim}
\subsection{Validation Rules}
\label{sec:org93c34bd}
\begin{itemize}
\item Validate before any calculation:
\begin{itemize}
\item presence of all required fields
\item exact field format compliance
\item type compliance
\item numeric range compliance
\item IANA-compliant timezone format
\item Gregorian date validity
\item 24-hour minute time validity
\end{itemize}
\item Strict numeric typing:
\begin{itemize}
\item numeric fields must be numeric values, not strings
\item no silent coercion of numeric-looking strings
\end{itemize}
\item No optional canonical input fields in v1.
\item Extra upstream fields are non-authoritative and must not affect results.
\end{itemize}
\subsection{Normalization Assumptions}
\label{sec:org6493c92}
\begin{itemize}
\item Before engine submission:
\begin{itemize}
\item location already resolved to unique latitude and longitude
\item timezone already resolved to a valid IANA identifier
\item date already resolved to Gregorian YYYY-MM-DD
\item time already normalized to HH:MM
\item ambiguity, alias handling, and UI cleaning already completed upstream
\item payload already reduced to one birth event
\end{itemize}
\end{itemize}
\subsection{Rejection Rules}
\label{sec:org8ced39f}
\begin{itemize}
\item \texttt{incomplete\_input}
\begin{itemize}
\item missing one or more required fields
\end{itemize}
\item \texttt{invalid\_input}
\begin{itemize}
\item malformed date or time
\item invalid timezone string
\item out-of-range latitude or longitude
\item numeric field supplied as string
\end{itemize}
\item \texttt{unsupported\_input}
\begin{itemize}
\item structurally valid input outside supported v1 scope
\end{itemize}
\end{itemize}
\subsection{No Silent Repair}
\label{sec:org8bc81f6}
\begin{itemize}
\item The engine must not silently:
\begin{itemize}
\item infer missing fields
\item correct malformed values
\item coerce ambiguous inputs
\item map timezone aliases
\item resolve place names
\item truncate unexpected time formats
\item substitute defaults
\end{itemize}
\end{itemize}
\section{Calculation Canon}
\label{sec:orga339fec}
\subsection{Shared Pre-Calculation Steps}
\label{sec:org61486c3}
\begin{itemize}
\item Validate input.
\item Construct local birth datetime from date, time, timezone.
\item Convert to UTC using the pinned timezone standard.
\item Derive Julian Day or equivalent astronomical time input.
\item Use provided latitude and longitude as authoritative coordinates.
\end{itemize}
\subsection{Angular Mechanics}
\label{sec:org879c29f}
\begin{itemize}
\item Canonical longitude range: \texttt{[0.0, 360.0)}.
\item Normalize by modular arithmetic.
\item \texttt{360.0} normalizes to \texttt{0.0}.
\item Opposition derivation uses canonical 180-degree opposition logic.
\item Cyclic interval handling must be deterministic.
\end{itemize}

\item House calculation uses:

\end{itemize}
\end{itemize}
\subsection{Human Design Design-Time Derivation}
\label{sec:org668f1b0}
\begin{itemize}
\item Design time is a second astronomical moment before birth.
\item Design-time target rule:
\begin{itemize}
\item find the prior moment where Sun longitude is exactly 88.0 degrees earlier than at birth
\item target longitude = normalize(birth\textsubscript{sun}\textsubscript{longitude} - 88.0)
\end{itemize}
\item Search direction:
\begin{itemize}
\item backward in time only
\end{itemize}
\item Initial bracket:
\begin{itemize}
\item centered around approximately 88 days before birth
\end{itemize}
\item Root-finding method:
\begin{itemize}
\item iterative bisection
\end{itemize}
\item Stop conditions:
\begin{itemize}
\item absolute Sun-longitude difference from target < 0.0001 degrees
\item or remaining interval width < 1 second
\end{itemize}
\item Forbidden shortcuts:
\begin{itemize}
\item fixed 88-day subtraction
\item fixed 3-month subtraction
\item rough platform imitation
\item average ephemeris approximation
\item precomputed lookup shortcut
\end{itemize}
\end{itemize}
\subsection{Human Design Gate And Line Mapping}
\label{sec:org09af851}
\begin{itemize}
\item Mandala parameters:
\begin{itemize}
\item anchor defined in Document 06
\item gate width = 5.625 degrees
\item line width = 0.9375 degrees
\item direction = ascending ecliptic longitude
\item intervals are start-inclusive and end-exclusive
\item gate order is a fixed lookup list
\end{itemize}
\item Relative mandala position:
\begin{itemize}
\item normalize longitude
\item subtract canonical anchor
\item normalize result into \texttt{[0.0, 360.0)}
\end{itemize}
\item Gate assignment:
\begin{itemize}
\item floor(relative\textsubscript{position} / 5.625)
\item use result as zero-based index into canonical gate order
\end{itemize}
\item Line assignment:
\begin{itemize}
\item compute offset inside gate segment
\item floor(offset / 0.9375)
\item convert to line number 1..6
\end{itemize}
\item Boundary rule:
\begin{itemize}
\item an interval end belongs to the next segment
\end{itemize}
\item Gene Keys uses exactly the same gate and line results as Human Design.
\end{itemize}

\begin{itemize}
\item canon review
\item canon revision where applicable
\item regression updates where applicable
\item version impact decision
\end{itemize}
\item Version changes:
\begin{itemize}
\item canon\textsubscript{version} changes for scope, calculation, mapping, output, precedence, formatting, or input behavior changes
\item input\textsubscript{schema}\textsubscript{version} changes for field/format/type/validation changes
\item mapping\textsubscript{version} changes for gate order, channel table, center list, channel-to-center mapping, identifier schemes, or deterministic mapping changes
\item source\textsubscript{stack}\textsubscript{version} changes for authoritative external source changes
\item engine\textsubscript{version} changes for any production implementation change
\end{itemize}
\end{itemize}
